--Log onto DAS3

--Create a file (helloworld.c) and type the following contents into helloworld.c 
====================================================================
#include "mpi.h"
#include <stdio.h>

int np, rank;

int main(int argc, char **argv)
{
  MPI_Init(&argc, &argv);
  MPI_Comm_size(MPI_COMM_WORLD, &np);
  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
  
  printf("Node %d of %d says: Hello world!\n", rank, np);
  
  MPI_Finalize();
  return 0;
}
====================================================================

--Load the three modules
====================================================================
module load prun
module load mpich
module load default-ethernet
====================================================================

--Compile the helloworld program
====================================================================
mpicc -o helloworld helloworld.c
====================================================================

--Run the helloworld program with 2 processes
====================================================================
prun -v -pbs-script /usr/local/sitedep/reserve.sge/sge-script `pwd`/helloworld 2
